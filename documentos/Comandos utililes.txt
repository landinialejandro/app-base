Ejecuta este comando desde la raíz del proyecto:

(
find app/Models -type f -name "*.php";
find app/Http/Controllers -type f -name "*.php";
find app/Http/Middleware -type f -name "*.php";
find database/migrations -type f -name "*.php";
find resources/views/orders -type f -name "*.blade.php";
find resources/views/documents -type f -name "*.blade.php";
) | sort | while read file; do
    echo "===== $file ====="
    cat "$file"
    echo
    echo
done > auditoria-tenant.txt


Para crear los archivos de vista:
touch resources/views/documents/index.blade.php
touch resources/views/documents/show.blade.php
touch resources/views/documents/create.blade.php
touch resources/views/documents/edit.blade.php
touch resources/views/documents/_form.blade.php

(
find app/Models -type f -name "Product.php";
find app/Http/Controllers -type f -name "ProductController.php";
find resources/views/products -type f -name "*.blade.php";
) | sort | while read file; do
    echo "===== $file ====="
    cat "$file"
    echo
    echo
done > auditoria-product.txt
